    public void reset() 
    { 
        if (ScreenState == FoodDrinkSelection) { 
            malteaser = 0; 
            MM = 0; 
            haribo = 0; 
            foodTotal = 0; 
            total -= foodTotal; 
        } 
        else 
        { 
            ticketCount = 0; 
            adultTickets = 0; 
            studentTickets = 0; 
            childTickets = 0; 
            ticketTotal = 0; 
            total = 0; 
        } 
        for( int i = 0; i < 3; i ++) 
        { 
            drink[i] = 0; 
            popcorn[i] = 0; 
        } 
    }
